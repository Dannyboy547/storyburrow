<script>

let selections = {
  character: "Dragon",
  world: "Enchanted Forest",
  mood: "Calm bedtime",
  type: "Night",
  length: "Short"
};

function select(el, type) {
  const parent = el.parentElement;
  Array.from(parent.children).forEach(c => c.classList.remove("active"));
  el.classList.add("active");
  selections[type] = el.innerText;
}

async function generateStory() {

  const btn = document.getElementById("generateBtn");

  if (btn.disabled) return;

  btn.disabled = true;
  btn.innerText = "🌙 Creating your story...";
  document.getElementById("loadingText").innerText = "✨ Building your magical story...";

  try {

    const name = document.getElementById("name").value || "A child";
    const companion = document.getElementById("companion").value || "a friend";

    const prompt = `
Write a high-quality personalised children's story.

The child’s name is: ${name}
Companion: ${companion}
Favourite character: ${selections.character}
World: ${selections.world}
Mood: ${selections.mood}
Time: ${selections.type}
Length: ${selections.length}

Include:
- Title at top
- Dialogue
- Beginning, middle, end
`;

    const response = await fetch("/api/story", {
      method: "POST",
      headers: {
        "Content-Type": "application/json"
      },
      body: JSON.stringify({
        name,
        companion,
        selections
      })
    });

    const data = await response.json();

    const story = data.story;

    const parts = story.split("\n\n");

    document.getElementById("storyTitle").innerText = parts[0].replace(/[#*]/g, "");
    document.getElementById("storyText").innerText = parts.slice(1).join("\n\n");

    document.getElementById("storyCard").classList.remove("hidden");

    document.getElementById("storyCard").scrollIntoView({ behavior: "smooth" });

  } catch (err) {
    alert("Something went wrong. Try again.");
  }

  btn.disabled = false;
  btn.innerText = "Create Story";
  document.getElementById("loadingText").innerText = "";
}

</script>
