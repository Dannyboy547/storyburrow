function saveStory() {
  const story = {
    title: storyTitle.innerText,
    text: storyText.innerText
  };

  let saved = JSON.parse(localStorage.getItem("stories") || "[]");
  saved.push(story);
  localStorage.setItem("stories", JSON.stringify(saved));
  alert("Story saved!");
}

function downloadStory() {
  const text = storyTitle.innerText + "\n\n" + storyText.innerText;

  const blob = new Blob([text], { type: "text/plain" });
  const url = URL.createObjectURL(blob);

  const a = document.createElement("a");
  a.href = url;
  a.download = "story.txt";
  a.click();

  URL.revokeObjectURL(url);
}

function viewStories() {
  let saved = JSON.parse(localStorage.getItem("stories") || "[]");
  const list = savedStoriesList;
  list.innerHTML = "";

  if (!saved.length) {
    list.innerHTML = "<p>No saved stories yet.</p>";
  } else {
    saved.forEach((s, i) => {
      const div = document.createElement("div");
      div.innerHTML = `
        <strong>${s.title}</strong><br><br>
        <button onclick="loadStory(${i})" class="primary-btn">📖 Read</button>
        <button onclick="deleteStory(${i})" class="primary-btn">🗑 Delete</button>
      `;
      list.appendChild(div);
    });
  }

  libraryCard.classList.remove("hidden");
}

function loadStory(i) {
  let saved = JSON.parse(localStorage.getItem("stories") || "[]");
  storyTitle.innerText = saved[i].title;
  storyText.innerText = saved[i].text;
  storyCard.classList.remove("hidden");
}

function deleteStory(i) {
  let saved = JSON.parse(localStorage.getItem("stories") || "[]");
  saved.splice(i, 1);
  localStorage.setItem("stories", JSON.stringify(saved));
  viewStories();
}

function readStory() {
  speechSynthesis.cancel();

  const text = storyTitle.innerText + ". " + storyText.innerText;
  if (!text) return;

  const speech = new SpeechSynthesisUtterance(text);
  speech.rate = 0.85;
  speech.pitch = 1;

  function speakNow() {
    const voices = speechSynthesis.getVoices();

    if (voices.length > 0) {
      const femaleVoice = voices.find(v =>
        v.name.toLowerCase().includes("female") ||
        v.name.includes("Google UK English Female")
      );

      speech.voice = femaleVoice || voices.find(v => v.lang.includes("en")) || voices[0];
      speechSynthesis.speak(speech);
    } else {
      setTimeout(speakNow, 100);
    }
  }

  speakNow();
}

function pauseStory() { speechSynthesis.pause(); }
function resumeStory() { speechSynthesis.resume(); }
function stopStory() { speechSynthesis.cancel(); }
