<script>

let selections = {
  character: "Dragon",
  world: "Enchanted Forest",
  mood: "Calm bedtime",
  type: "Night",
  length: "Short"
};

function select(el, type) {
  const parent = el.parentElement;
  Array.from(parent.children).forEach(c => c.classList.remove("active"));
  el.classList.add("active");
  selections[type] = el.innerText.split(" (")[0];
}

function saveStory() {
  const story = {
    title: storyTitle.innerText,
    text: storyText.innerText
  };

  let saved = JSON.parse(localStorage.getItem("stories") || "[]");
  saved.push(story);
  localStorage.setItem("stories", JSON.stringify(saved));

  alert("Story saved!");
}

function viewStories() {
  let saved = JSON.parse(localStorage.getItem("stories") || "[]");
  const list = savedStoriesList;
  list.innerHTML = "";

  if (!saved.length) {
    list.innerHTML = "<p>No saved stories yet.</p>";
  } else {
    saved.forEach((s, i) => {
      const div = document.createElement("div");
      div.innerHTML = `
        <strong>${s.title}</strong><br><br>
        <button onclick="loadStory(${i})" class="primary-btn">📖 Read</button>
        <button onclick="deleteStory(${i})" class="primary-btn">🗑 Delete</button>
      `;
      list.appendChild(div);
    });
  }

  libraryCard.classList.remove("hidden");
}

function loadStory(i) {
  let saved = JSON.parse(localStorage.getItem("stories") || "[]");
  storyTitle.innerText = saved[i].title;
  storyText.innerText = saved[i].text;
  storyCard.classList.remove("hidden");
}

function deleteStory(i) {
  let saved = JSON.parse(localStorage.getItem("stories") || "[]");
  saved.splice(i, 1);
  localStorage.setItem("stories", JSON.stringify(saved));
  viewStories();
}

function readStory() {
  const speech = new SpeechSynthesisUtterance(storyText.innerText);
  speech.rate = 0.75;
  speech.pitch = 1.1;
  speechSynthesis.speak(speech);
}

async function generateStory() {

  const btn = document.getElementById("generateBtn");
  const loading = document.getElementById("loadingText");

  if (btn.disabled) return;

  btn.disabled = true;
  btn.innerText = "🌙 Creating your story...";

  const messages = [
    "✨ Creating your magical story...",
    "🌙 Sprinkling a little bedtime magic...",
    "📖 Writing something special..."
  ];

  let i = 0;
  const interval = setInterval(() => {
    loading.innerText = messages[i % messages.length];
    i++;
  }, 1500);

  try {

    const name = document.getElementById("name").value.trim() || "A child";
    const companion = document.getElementById("companion").value.trim() || "a friend";

    const response = await fetch("/api/story", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        name,
        companion,
        selections,
        length: selections.length
      })
    });

    const data = await response.json();

    const story = data.story || "Story failed to generate.";

    const parts = story.split("\n\n");

    document.getElementById("storyTitle").innerText = parts[0].replace(/[#*]/g, "");
    document.getElementById("storyText").innerText = parts.slice(1).join("\n\n");

    document.getElementById("storyCard").classList.remove("hidden");

    setTimeout(() => {
      document.getElementById("storyTitle").scrollIntoView({ behavior: "smooth" });
    }, 300);

  } catch (err) {
    alert("Something went wrong. Try again.");
    console.error(err);
  }

  clearInterval(interval);
  loading.innerText = "";

  btn.disabled = false;
  btn.innerText = "Create Story";
}

</script>
